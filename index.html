    import java.io.BufferedReader;
    import java.io.IOException;
    import java.io.InputStreamReader;

    public class Reader {
            public static Chivalry getCheck() throws IOException {

            BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
            String coordinates;

            while (true) {
                System.out.print("Enter number of cell: ");
                coordinates = reader.readLine();

                if ((coordinates.length() < 3) && (Constants.ALPHABET.contains(coordinates.substring(0, 1))
                                               && (Constants.NUMERIC.contains(coordinates.substring(1))))) {
                    System.out.println("Correct input");
                    return new Chivalry(Constants.ALPHABET.indexOf(coordinates.substring(0, 1)) + 1, 
                                        Constants.NUMERIC.indexOf(coordinates.substring(1)) + 1);
                }
                else System.out.println("Incorrect input, try again\n");
            }
        }
    }
